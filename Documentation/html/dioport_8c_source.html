118 {

121 
122  if (AddressEnd > 0xffff)
123  AddressEnd = 0xffff;
124 
125  if (Address <= AddressEnd)

437 {
438  PIO_STACK_LOCATION IoStackLocation;
439  ULONG IoControlCode;

443  PEPROCESS CurrentProcess;
444

446 
447 // DIO_IN_DEBUG_BREAKPOINT();
448 
449  CurrentProcess = PsGetCurrentProcess();
450  Status = STATUS_SUCCESS;
451 
452  IoStackLocation = IoGetCurrentIrpStackLocation(Irp);
453  IoControlCode = IoStackLocation->Parameters.DeviceIoControl.IoControlCode;
454  InputBufferLength = IoStackLocation->Parameters.DeviceIoControl.InputBufferLength;
455 
456  DIO_FUNC_TRACE("IOCTL from process %d\n", PsGetProcessId(CurrentProcess));
457 
458  do
459  {
460  //
461  // 1. Make sure that we're using buffered IOCTL.
462  //
463 
464  if (METHOD_FROM_CTL_CODE(IoControlCode) != METHOD_BUFFERED)
465  {
466  Status = STATUS_NOT_SUPPORTED;
467  break;
468  }
469 
470 
471  //
472  // 2. Validate the buffer.
473  //
474 
475  Packet = (DIO_PACKET *)Irp->AssociatedIrp.SystemBuffer;
476  if (!DiopValidatePacketBuffer(Packet, InputBufferLength, IoControlCode))
477  {
478  DIO_FUNC_TRACE("Buffer validation failed\n");
479  Status = STATUS_INVALID_PARAMETER;
480  break;
481  }
482 
483 
484  //
485  // 3. Dispatch IOCTL request.
486  //
487 
488  switch (IoControlCode)
489  {
490  case DIO_IOCTL_SET_PORTACCESS:
491  if (!DioRegisterSelf())
492  {
493  DIO_FUNC_TRACE("Process 0x%p is already registered\n", DiopRegisteredProcess);
494  Status = STATUS_ACCESS_DENIED;
495  break;
496  }
497 
498  // Create the I/O access map from packet.
499  if (!DioSetIoAccessMapByPacket(&Packet->PortAccess, DiopIoAccessMap))
500  {
501  DIO_FUNC_TRACE("Failed to prepare access map\n");
502  Status = STATUS_INVALID_PARAMETER;
503  break;
504  }
505 
506  DIO_FUNC_TRACE("Setting port permission for process 0x%p (%d)\n",
507  CurrentProcess, PsGetProcessId(CurrentProcess));
508 
509  Ke386IoSetAccessProcess(CurrentProcess, 1);
510  Ke386SetIoAccessMap(1, DiopIoAccessMap);
511  break;
512 
513  case DIO_IOCTL_RESET_PORTACCESS:
514  if (!DioSetIoAccessMapByPacket(NULL, DiopIoAccessMap))
515  {
516  DIO_FUNC_TRACE("Failed to prepare access map\n");
517  Status = STATUS_INVALID_PARAMETER;
518  break;
519  }
520 
521  DIO_FUNC_TRACE("Resetting port permission to initial state for process 0x%p (%d)\n",
522  CurrentProcess, PsGetProcessId(CurrentProcess));
523 
524  Ke386IoSetAccessProcess(CurrentProcess, 0);
525  break;
526 
527  default:
528  Status = STATUS_NOT_SUPPORTED;
529  }
530  } while(FALSE);
531 
532 
533  //
534  // 4. Complete the request.
535  //
536 
537  Irp->IoStatus.Status = Status;
538  Irp->IoStatus.Information = 0;
539 
540  IofCompleteRequest(Irp, IO_NO_INCREMENT);
541 
542 
543  return STATUS_SUCCESS;
544 }
